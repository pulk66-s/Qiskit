\title{Inner product}

\section{Definition}
\begin{equation}
    (|\psi\rangle,|\phi\rangle)
\end{equation}
Where $|\psi\rangle$ and $|\phi\rangle$ are vectors in a Vector space.

\section{Properties}

Right linearity
\begin{equation}
    (|\psi\rangle,|\phi\rangle) = (|\psi\rangle,\sum_{i}\lambda_i|\phi_i\rangle)
        = \sum_{i}\lambda_i(|\psi\rangle,|\phi_i\rangle)
\end{equation}
