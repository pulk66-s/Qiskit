\section{Find x, y and z such that $(x - y, x + y, z - 1) = (4, 2, 3)$}
\begin{equation}
    \begin{split}
        & \begin{cases}
            x - y & = 4 \\
            x + y & = 2 \\
            z - 1 & = 3 \\
        \end{cases} \\
        & \begin{cases}
            x & = 4 + y     \\
            y + 4 + y & = 2 \\
            z & = 4         \\
        \end{cases} \\
        & \begin{cases}
            x & = 4 + y \\
            y & = -1    \\
            z & = 4     \\
        \end{cases} \\
        & \begin{cases}
            x & = 3     \\
            y & = -1    \\
            z & = 4     \\
        \end{cases} \\
        \Longrightarrow (x - y, x + y, z - 1) & = (4, 2, 3), x = 3, y = -1, z = 4
    \end{split}
\end{equation}
